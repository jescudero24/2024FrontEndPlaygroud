  const readline = require('readline')

  const rl = readline.createInterface({
      input: process.stdin,
      output: process.stdout
  })
  const lines = []
  rl.on('line', (input) => {
      lines.push(input);
  })
  rl.on('close', () => {
      let l = 0;
      let t = +lines[l++]
      const output = []
      for (let i = 0; i < t; i++) {
          const n = +lines[l++]
          const arr = lines[l++].trim().split(' ').map(Number)
          output[i] = solve(n, arr) ? 'YES' : 'NO'
      }
      console.log(output.join('\n'))
  })
  
  function solve(n, arr) {
      if (arr[0] !== arr[n - 1]) return false
      for (let i = 0; i < n; i++) {
          if (arr[i] < arr[0]) return false
      }
      return true
  }
